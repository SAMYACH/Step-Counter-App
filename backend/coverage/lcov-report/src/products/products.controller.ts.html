import { Body, CacheInterceptor, CacheTTL, Controller, Delete, Get, HttpStatus, Inject, Param, ParseIntPipe, Post, Put, UseFilters, UseInterceptors, UsePipes, ValidationPipe } from '@nestjs/common';
import { ApiTags, ApiOkResponse, ApiInternalServerErrorResponse, ApiCreatedResponse, ApiResponse, ApiNotFoundResponse } from '@nestjs/swagger';
 
import { ProductsService } from './products.service';
import { ProductDTO } from './Productdto/Product.dto';
import { HttpExceptionFilter } from '../common/exception.filter';
import { Console } from 'console';
/**
 * Product Controller responds to inbound HTTP Requests and produces HTTP Responses.
 */
@UseInterceptors(CacheInterceptor)
//@UseFilters(new HttpExceptionFilter())
//@UsePipes(new ValidationPipe())
@ApiTags('product')
@Controller('product')
export class ProductsController {
  constructor(private productServices: ProductsService) {
 
  }
  /**
    * for fetch all record from Products
    * @returns  list of Products
    */
  @ApiOkResponse({ status: HttpStatus.OK, description: 'Product Fetched Sucessfully' })
  @ApiInternalServerErrorResponse({ status: HttpStatus.INTERNAL_SERVER_ERROR, description: 'This is a internel server error.try again later ' })
  @ApiNotFoundResponse({ status: HttpStatus.NOT_FOUND, description: "no Data found" })
  @UseFilters(new HttpExceptionFilter())
  @CacheTTL(30)
  @Get()
  async fetchAllroduct() {
    console.log("In Controller")
    return await this.productServices.fetchAllroduct()
  }
  /**
    * create new Products
    * @param ProductDTO specify Products name
    * @returns newly created Products
    */
  @ApiOkResponse({ status: HttpStatus.CREATED, description: 'Products Added Sucessfully' })
  @ApiInternalServerErrorResponse({ status: HttpStatus.INTERNAL_SERVER_ERROR, description: 'This is a internel server error.try again later ' })
  @Post()
  async createProduct(@Body() productDTO:ProductDTO) {
    return await this.productServices.addroduct(productDTO)
  }
  /**
 * list Products based on id
 * @param id Products id
 * @returns Products
 */
  @ApiCreatedResponse({ status: HttpStatus.OK, description: 'Products Fetched Sucessfully for id' })
  @ApiInternalServerErrorResponse({ status: HttpStatus.INTERNAL_SERVER_ERROR, description: 'This is a internel server error.try again later ' })
  @ApiNotFoundResponse({ status: HttpStatus.NOT_FOUND, description: "no Data found for given id" })
  @UseFilters(new HttpExceptionFilter())
  @Get(':id')
  async getbyProductid(@Param('id') id: number): Promise<ProductDTO> {
    console.log("in controller")
    return await this.productServices.getbyroductid(id);
  }
  /**
* update Products based on category
* @param ProductDTO update 
* @param id Products id
* @returns updated
*/
  @ApiOkResponse({ status: HttpStatus.OK, description: 'Updated Sucessfully' })
  @ApiInternalServerErrorResponse({ status: HttpStatus.INTERNAL_SERVER_ERROR, description: 'This is a internel server error.try again later ' })
 
  @Put(':id')
 
  updateProduct(@Body() ProductDTO: ProductDTO, @Param('id') id: number): Promise<ProductDTO> {
    return this.productServices.updateroduct(ProductDTO, id)
    //return ProductDTO;
  }
  /**
     * delete Products based on id
     * @param id Products id
     * @returns no of effected rows
     */
  @ApiOkResponse({ status: HttpStatus.OK, description: 'Deleted Sucessfully' })
  @ApiNotFoundResponse({ status: HttpStatus.NOT_FOUND, description: 'Products Not Found' })
  @ApiInternalServerErrorResponse({ status: HttpStatus.INTERNAL_SERVER_ERROR, description: 'This is a internel server error.try again later ' })
 
  @Delete(':id')
 
  deleteProduct(@Param('id') id: number) {
    return this.productServices.deleteProduct(id);
  }
  /**
* get all  Products based with category
* @param id Products id
* @returns no of effected rows
*/
  @Get('all/category')
  async getProductall() {
    return this.productServices.getProductall();
 
  }
  /**
         *save  Products based with category
         * @param id Products id
         * @returns no of effected rows
         */
  @Post('/productwithcategory')
  async addProductWithCategory(@Body() ProductDTO: ProductDTO) :Promise<ProductDTO>{
    return await this.productServices.addProductWithCategory(ProductDTO)
  }
}
